🏗️ Schema & DDL

schema.sql + sale-migration.sql
CREATE Create Database schema.sql 
CREATE DATABASE IF NOT EXISTS property_mgmt
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;
CREATE AGENT table schema.sql 
CREATE TABLE IF NOT EXISTS AGENT (
  Agent_ID        INT AUTO_INCREMENT PRIMARY KEY,
  Name            VARCHAR(100) NOT NULL,
  Phone           VARCHAR(20)  NOT NULL,
  Commission_Rate DECIMAL(5,2) NOT NULL,
  CONSTRAINT chk_commission
    CHECK (Commission_Rate BETWEEN 0 AND 25)
);
CREATE PROPERTY table schema.sql 
CREATE TABLE IF NOT EXISTS PROPERTY (
  Property_ID  INT AUTO_INCREMENT PRIMARY KEY,
  Title        VARCHAR(200) NOT NULL,
  Type         VARCHAR(50)  NOT NULL,
  Location     VARCHAR(255) NOT NULL,
  Price        DECIMAL(14,2) NOT NULL,
  Status       VARCHAR(20)  DEFAULT 'Available',
  AI_Est_Price DECIMAL(14,2) NULL,
  Owner_ID     INT NOT NULL,
  Agent_ID     INT NOT NULL,
  CONSTRAINT chk_price   CHECK (Price > 0),
  CONSTRAINT chk_status  CHECK (Status IN
    ('Available','Sold','Rented','Pending'))
);
CREATE SALE table (migration) sale-migration.sql 
CREATE TABLE IF NOT EXISTS SALE (
  Sale_ID        INT AUTO_INCREMENT PRIMARY KEY,
  Property_ID    INT NOT NULL,
  Buyer_Tenant_ID INT NOT NULL,
  Amount         DECIMAL(15,2) NOT NULL,
  Method         VARCHAR(30) DEFAULT NULL,
  Sale_Status    ENUM('Pending_Payment','Completed',
                      'Cancelled') DEFAULT 'Pending_Payment',
  Created_At     TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

📊 Dashboard

routes/dashboard.js
SELECT Count Totals (Properties, Agents, Tenants, Owners) GET /api/dashboard 
SELECT COUNT(*) AS total_properties FROM PROPERTY;
SELECT COUNT(*) AS total_agents    FROM AGENT;
SELECT COUNT(*) AS total_tenants   FROM TENANT;
SELECT COUNT(*) AS total_owners    FROM OWNER;
SELECT Active Leases & Open Inquiries GET /api/dashboard 
SELECT COUNT(*) AS active_leases
FROM LEASE WHERE End_Date >= CURDATE();

SELECT COUNT(*) AS open_inquiries
FROM INQUIRY
WHERE Status IN ('New', 'Responded');
SELECT Total Revenue & Available Properties GET /api/dashboard 
SELECT IFNULL(SUM(Amount), 0) AS total_revenue
FROM PAYMENT WHERE Status = 'Success';

SELECT COUNT(*) AS available_props
FROM PROPERTY WHERE Status = 'Available';
SELECT Property Status Breakdown GET /api/dashboard 
SELECT Status, COUNT(*) AS count
FROM PROPERTY
GROUP BY Status;
SELECT Recent Payments (last 5) GET /api/dashboard 
SELECT pay.*, p.Title AS Property_Title
FROM PAYMENT pay
JOIN LEASE l    ON pay.Lease_ID = l.Lease_ID
JOIN PROPERTY p ON l.Property_ID = p.Property_ID
ORDER BY pay.Payment_Date DESC LIMIT 5;
SELECT Top Agents by Property Count GET /api/dashboard 
SELECT a.Name, COUNT(p.Property_ID) AS property_count
FROM AGENT a
LEFT JOIN PROPERTY p ON a.Agent_ID = p.Agent_ID
GROUP BY a.Agent_ID
ORDER BY property_count DESC LIMIT 5;

📄 Leases

SELECT All Leases with Tenants (GROUP_CONCAT) GET /api/leases 
SELECT l.*, p.Title AS Property_Title,
  GROUP_CONCAT(t.Name SEPARATOR ', ') AS Tenants
FROM LEASE l
JOIN      PROPERTY p  ON l.Property_ID = p.Property_ID
LEFT JOIN TENANT_LEASE tl ON l.Lease_ID = tl.Lease_ID
LEFT JOIN TENANT t    ON tl.Tenant_ID = t.Tenant_ID
GROUP BY l.Lease_ID
ORDER BY l.Start_Date DESC;

SELECT All Inquiries (Tenant + Property + Agent) GET /api/inquiries 
SELECT i.*,
  t.Name AS Tenant_Name,
  p.Title AS Property_Title,
  a.Name AS Agent_Name
FROM INQUIRY i
JOIN TENANT   t ON i.Tenant_ID   = t.Tenant_ID
JOIN PROPERTY  p ON i.Property_ID = p.Property_ID
JOIN AGENT     a ON i.Agent_ID    = a.Agent_ID
ORDER BY i.Date DESC;
