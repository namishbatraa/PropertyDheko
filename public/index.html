CREATECreate Databaseschema.sql
CREATE DATABASE IF NOT EXISTS property_mgmt
  CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
CREATEAGENT tableschema.sql
CREATE TABLE IF NOT EXISTS AGENT (
  Agent_ID INT AUTO_INCREMENT PRIMARY KEY,
  Name     VARCHAR(100) NOT NULL,
  Phone    VARCHAR(20)  NOT NULL,
  Commission_Rate DECIMAL(5,2) NOT NULL,
  CONSTRAINT chk_commission CHECK (Commission_Rate BETWEEN 0 AND 25)
);
CREATEPROPERTY tableschema.sql
CREATE TABLE IF NOT EXISTS PROPERTY (
  Property_ID INT AUTO_INCREMENT PRIMARY KEY,
  Title VARCHAR(200) NOT NULL, Type VARCHAR(50) NOT NULL,
  Location VARCHAR(255) NOT NULL, Price DECIMAL(14,2) NOT NULL,
  Status VARCHAR(20) DEFAULT 'Available', AI_Est_Price DECIMAL(14,2),
  Owner_ID INT NOT NULL, Agent_ID INT NOT NULL
);
CREATESALE table (migration)sale-migration.sql
CREATE TABLE IF NOT EXISTS SALE (
  Sale_ID INT AUTO_INCREMENT PRIMARY KEY,
  Property_ID INT NOT NULL, Buyer_Tenant_ID INT NOT NULL,
  Amount DECIMAL(15,2) NOT NULL, Method VARCHAR(30),
  Sale_Status ENUM('Pending_Payment','Completed','Cancelled')
    DEFAULT 'Pending_Payment'
);

SELECTCount TotalsGET /api/dashboard
SELECT COUNT(*) AS total_properties FROM PROPERTY;
SELECT COUNT(*) AS total_agents    FROM AGENT;
SELECT COUNT(*) AS total_tenants   FROM TENANT;
SELECT COUNT(*) AS total_owners    FROM OWNER;
SELECTActive Leases & Open InquiriesGET /api/dashboard
SELECT COUNT(*) AS active_leases FROM LEASE
WHERE End_Date >= CURDATE();
SELECT COUNT(*) AS open_inquiries FROM INQUIRY
WHERE Status IN ('New','Responded');
SELECTTotal RevenueGET /api/dashboard
SELECT IFNULL(SUM(Amount),0) AS total_revenue
FROM PAYMENT WHERE Status = 'Success';
SELECTStatus BreakdownGET /api/dashboard
SELECT Status, COUNT(*) AS count
FROM PROPERTY GROUP BY Status;
SELECTRecent Payments (last 5)GET /api/dashboard
SELECT pay.*, p.Title AS Property_Title
FROM PAYMENT pay
JOIN LEASE l ON pay.Lease_ID = l.Lease_ID
JOIN PROPERTY p ON l.Property_ID = p.Property_ID
ORDER BY pay.Payment_Date DESC LIMIT 5;
SELECTTop Agents by Property CountGET /api/dashboard
SELECT a.Name, COUNT(p.Property_ID) AS property_count
FROM AGENT a
LEFT JOIN PROPERTY p ON a.Agent_ID = p.Agent_ID
GROUP BY a.Agent_ID
ORDER BY property_count DESC LIMIT 5;

📄 Leases routes/leases.js
SELECTAll Leases with Tenants (GROUP_CONCAT)GET /api/leases
SELECT l.*, p.Title AS Property_Title,
  GROUP_CONCAT(t.Name SEPARATOR ', ') AS Tenants
FROM LEASE l
JOIN PROPERTY p ON l.Property_ID = p.Property_ID
LEFT JOIN TENANT_LEASE tl ON l.Lease_ID = tl.Lease_ID
LEFT JOIN TENANT t ON tl.Tenant_ID = t.Tenant_ID
GROUP BY l.Lease_ID ORDER BY l.Start_Date DESC;

💬 Inquiries routes/inquiries.js
SELECTAll Inquiries (4-table JOIN)GET /api/inquiries
SELECT i.*, t.Name AS Tenant_Name,
  p.Title AS Property_Title, a.Name AS Agent_Name
FROM INQUIRY i
JOIN TENANT t   ON i.Tenant_ID=t.Tenant_ID
JOIN PROPERTY p ON i.Property_ID=p.Property_ID
JOIN AGENT a   ON i.Agent_ID=a.Agent_ID
ORDER BY i.Date DESC;
